CREATE DATABASE IF NOT EXISTS twittercopy;

USE twittercopy;

CREATE TABLE IF NOT EXISTS customers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL,
    playerLevel FLOAT NOT NULL,
    createDate TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


SELECT C.userID, C.vorname, C.nachname, A.abteilungsname FROM `customers` as C INNER JOIN abteilungen as A ON C.abteilungsID = A.abteilungsID
